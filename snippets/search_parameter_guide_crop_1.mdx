```rust Rust
let results: SearchResults<Movie> = client
  .index("movies")
  .search()
  .with_query("shifu")
  .with_attributes_to_crop(Selectors::Some(&[("overview", None)]))
  .with_crop_length(5)
  .execute()
  .await
  .unwrap();

// Get the formatted results
let formatted_results: Vec<&Movie> = results
  .hits
  .iter()
  .map(|r| r.formatted_result.as_ref().unwrap())
  .collect();
```
